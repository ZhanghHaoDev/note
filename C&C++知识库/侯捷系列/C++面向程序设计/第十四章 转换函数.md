# 第十四章 转换函数

# 1. 数据类型重载

class Fraction

{

public:

Fraction(int num,int den = 1)

: m_numberstor(num),m_denominator(den)

operator double() const {

return m_numberstor / m_denominator;

}

private:

int m_numberstor;   // 分子

int m_denominator;  // 分母

};

---

Fraction f(3,5);

double d = 4 + f;

结果是：4.6
